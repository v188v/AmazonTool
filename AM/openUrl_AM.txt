url=read.table("D:/Documents/RStudio/Work/Amazon/AM/URL/AMAZONurlAM_OPEN.csv", header = FALSE, sep = ",",stringsAsFactors=FALSE);

addr=function(x)
{
  length=nrow(x);
  tmp="";
  for (i in 1:length)
  {
   tmp[i]=x[i,1];
   }
   tmp;
}
url=addr(url);

open=function(x)
{
shell.exec(x)
}
